```json
{
  "type": "object",
  "properties": {
    "workdir": {
      "type": "string",
      "description": "Absolute path. Becomes SessionRoot per D-D2."
    },
    "model": {
      "type": "string",

      "type": "string",

  },
  "required": [
    "workdir"
  ]
}
```
